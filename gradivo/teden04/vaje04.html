Zapisali bomo program v programskem jeziku java, ki prikazuje razlike med podatkovnimi tipi.

public class SpremenljivkeTipi {
public static void main(String[] args) {

int stevilo; // Deklaracija
stevilo = 42;// Inicializacija

int spremenljivka = 4 * 8 + 20 / 2; // spremenljivka = izraz;

boolean logicnaVrednost = true; // boolean
byte steviloByte = 124; // byte
short steviloShort = -200; // short
int steviloInt = -4250000; // int
long steviloLong = 42332200000L; // long
float steviloFloat = -42.38f; // float
double steviloDouble = 832.48329; // double
char znak = 'a'; // char
char znakInt = 65; // char (int) = 'A'
String niz = "Programiranje je zabavno."; // String
}
}
